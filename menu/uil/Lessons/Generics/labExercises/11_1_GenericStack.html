11.1 GenericStack

Source Code: GenericStack

Description

Below is a set of classes that implement a Stack ADT. Modify the code so that the ADT uses generics. For example, a client program should be able to specify the data type of the stack when an object is instantiated.

GenericStack<String> stack = new GenericStack<String>();

Modify the following classes and interfaces so they become generic:

  • GenericStack class
  • ListNode class
  • Stack interface

GenericStack class

public class GenericStack implements Stack
{
    private ListNode head;
    
    public GenericStack()
    {
       head = null; 
    }
    
    // adds an item to the top of the stack
    public void push(Object item)
    {
       if(head == null)
       {
          head = new ListNode(item, null);
       }
       else
       {
           ListNode temp = new ListNode(item, head);
           head = temp;
       }    
    }
    
    // removes an item from the top of the stack
    public Object pop()
    {
       ListNode temp = head;
       head = head.getNext();
       return temp.getValue();
    }
    
    // returns the top item in the stack without removing it
    public Object top()
    {
        return head.getValue();
    }
    
    // returns true if the stack is empty; otherwise returns false
    public boolean isEmpty()
    {
       if(head == null)
         return true;
       else
         return false;  
    }
    
   public String toString()
   {
      String str="";
      ListNode cur = head;
      while(cur != null)
      {
         str += cur.getValue() + "\n";
         cur = cur.getNext();
      }
      return str;
   }
}

Stack interface

public interface Stack
{
    // adds an item to the top of the stack
    public void push(Object item);
    
    // removes an item from the top of the stack
    public Object pop();
    
    // returns the top item in the stack without removing it
    public Object top();
    
    // returns true if the stack is empty; otherwise returns false
    public boolean isEmpty();
    
    // returns a String representation of the stack
    public String toString();
    
}

ListNode class

// APSubset ListNode class

public class ListNode
{  
   // Instance Variables
   private Object value;
   private ListNode next;
  
   // Constructor
   public ListNode(Object initValue, ListNode initNext)
   { 
      value = initValue; 
      next = initNext; 
   }

   // Accessor Methods
   public Object getValue() 
   { 
      return value; 
   }
  
   public ListNode getNext() 
   { 
      return next; 
   }

   // Mutator Methods
   public void setValue(Object theNewValue)
   { 
      value = theNewValue; 
   }
  
   public void setNext(ListNode theNewNext)
   { 
      next = theNewNext;
   }
}
